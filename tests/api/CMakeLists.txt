# UFT API Tests
file(GLOB API_TEST_SOURCES "*.c")
if(API_TEST_SOURCES AND TARGET uft_core)
    foreach(test_src ${API_TEST_SOURCES})
        get_filename_component(test_name ${test_src} NAME_WE)
        add_executable(api_${test_name} ${test_src})
        target_link_libraries(api_${test_name} PRIVATE uft_core)
    endforeach()
else()
    message(STATUS "API Tests: Skipping")
endif()
